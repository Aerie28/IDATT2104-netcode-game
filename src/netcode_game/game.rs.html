19/// Game state that tracks all players and their positions, and ids for the players
20pub struct Game {
21    players: HashMap<SocketAddr, PlayerState>,
22    id_to_addr: HashMap<Uuid, SocketAddr>,
23    addr_to_id: HashMap<SocketAddr, Uuid>,
24    last_processed: HashMap<Uuid, u32>, // Track inputs
25}

39    /// Handles new connection by adding player at random pos/color
40    pub fn connect_player(&mut self, addr: SocketAddr) -> Uuid {
41        use rand::Rng;
42
43        // Check if player already connected
44        if self.players.contains_key(&addr) {
45            // Player already connected
46            return *self.addr_to_id.get(&addr).unwrap();
47        }
48
49        // Generate a random position within the board bounds
50        let mut rng = rand::rng();
51        let x = rng.random_range(PLAYER_SIZE..(BOARD_WIDTH - (PLAYER_SIZE)));
52        let y = rng.random_range(PLAYER_SIZE..(BOARD_HEIGHT - (PLAYER_SIZE) - TOOL_BAR_HEIGHT));
53        
54        // Pick a color from the palette randomly
55        let palette = player_colors::get_palette();
56        let color_base = palette[rng.random_range(0..palette.len())];
57        // Pack the color as u32 for serialization
58        let color = ((color_base.r * 255.0) as u32) << 16
59            | ((color_base.g * 255.0) as u32) << 8
60            | ((color_base.b * 255.0) as u32);
61
62        // Store the player ID
63        let id = Uuid::new_v4();
64        self.id_to_addr.insert(id, addr);
65        self.addr_to_id.insert(addr, id);
66
67        // Initialize player position and history
68        let initial_position = Position { x, y };
69        let mut position_history = Vec::with_capacity(MAX_POSITION_HISTORY);
70        position_history.push(PositionSnapshot {
71            position: initial_position,
72            timestamp: Instant::now().elapsed().as_millis() as u64,
73        });
74
75        // Insert the player state into the game
76        self.players.insert(
77            addr,
78            PlayerState {
79                position: initial_position,
80                color,
81                last_active: Instant::now(),
82                position_history,
83            },
84        );
85        id
86    }
87
88    /// Handle player input and update position + activity
89    pub fn handle_input(&mut self, addr: SocketAddr, input: PlayerInput) {
90        if let Some(player) = self.players.get_mut(&addr) {
91            player.last_active = Instant::now();
92
93            // Update last processed input
94            if let Some(id) = self.addr_to_id.get(&addr) {
95                self.last_processed.insert(*id, input.sequence);
96            }
97
98            // Update player position based on input direction for prediction
99            match input.dir {
100                Direction::Up => player.position.y = player.position.y.saturating_sub(PLAYER_SPEED).max(PLAYER_SIZE),
101                Direction::Down => player.position.y = player.position.y.saturating_add(PLAYER_SPEED).min(BOARD_HEIGHT - (PLAYER_SIZE) - TOOL_BAR_HEIGHT),
102                Direction::Left => player.position.x = player.position.x.saturating_sub(PLAYER_SPEED).max(PLAYER_SIZE),
103                Direction::Right => player.position.x = player.position.x.saturating_add(PLAYER_SPEED).min(BOARD_WIDTH - (PLAYER_SIZE)),
104            }
105
106            // Store current position in history
107            let current_time = Instant::now().elapsed().as_millis() as u64;
108            player.position_history.push(PositionSnapshot {
109                position: player.position,
110                timestamp: current_time,
111            });
112
113            // Keep only the last MAX_POSITION_HISTORY entries
114            if player.position_history.len() > MAX_POSITION_HISTORY {
115                player.position_history.remove(0);
116            }
117        }
118    }

120    /// Marks players inactive if timeout exceeded
121    pub fn update_server_dropped(&mut self) {
122        let now = Instant::now();
123        let mut to_disconnect = Vec::new();
124        
125        // Check for players that haven't sent a ping in TIMEOUT duration
126        for (addr, player) in self.players.iter() {
127            if now.duration_since(player.last_active) >= TIMEOUT {
128                to_disconnect.push(*addr);
129            }
130        }
131        
132        // Disconnect inactive players
133        for addr in to_disconnect {
134            if let Some(id) = self.addr_to_id.get(&addr) {
135                println!("Player {} disconnected due to timeout", id);
136            }
137            self.disconnect_player(&addr);
138        }
139    }

141    /// Get player address of active player
142    pub fn active_player_addrs(&self) -> Vec<SocketAddr> {
143        self.players.keys().cloned().collect()
144    }
145
146    /// Remove player on disconnect
147    pub fn disconnect_player(&mut self, addr: &SocketAddr) {
148        if let Some(id) = self.addr_to_id.remove(addr) {
149            self.id_to_addr.remove(&id);
150            self.last_processed.remove(&id);
151        }
152        self.players.remove(addr);
153    }
154
155    /// Build a snapshot of active players for broadcasting
156    pub fn build_snapshot(&self) -> GameState {
157        let players = self.players.iter()
158            .map(|(addr, p)| {
159                let player_id = *self.addr_to_id.get(addr).unwrap();
160                (player_id, p.position, p.color)
161            })
162            .collect();
163        GameState {
164            players,
165            last_processed: self.last_processed.clone(),
166            server_timestamp: Instant::now().elapsed().as_millis() as u64,
167        }
168    }
169
170    /// Mutable access to players (use only when necessary)
171    pub fn get_players_mut(&mut self) -> &mut HashMap<SocketAddr, PlayerState> {
172        &mut self.players
173    }
174}
